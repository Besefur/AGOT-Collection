character_event = {
	id = kingsmoot.9
	desc = "EVTDESCkingsmoot.9"
	picture = "GFX_drowned"

	is_triggered_only = yes

	trigger = { NOT = { has_character_flag = kingsmoot_candidate_contesting } }

	immediate = {
		if = { #for loc
			limit = { ai = no }
			any_liege = {
				limit = {
					primary_title = { has_law = succ_kingsmoot_elective }
				}
				save_event_target_as = kingsmoot_liege
			}
		}
		#set_character_flag = kingsmoot_candidate_contesting
		# clr_character_flag = large_gift
		# clr_character_flag = regular_gift
		# clr_character_flag = small_gift
		#clr_character_flag = no_gift
		clr_character_flag = threats
		clr_character_flag = sunset_sea
		clr_character_flag = peace
		#set_variable = { which = "election_votes" value = 0 }

		#AI calls in favors
		if = {
			limit = { ai = yes }
			any_liege = {
				limit = {
					primary_title = { has_law = succ_kingsmoot_elective }
					OR = { #Must be likely to die soon
						age = 60
						NOT = { health = 1 }
					}
				}
				primary_title = { save_event_target_as = kingsmoot_title }
				any_realm_character = {
					limit = {
						owes_favor_to = ROOT
						ai = yes
						OR = {
							can_press_claims_trigger = yes
							trait = drowned
						}
						OR = { #Must either follow Drowned God or be dejure part of the Kingdom
							is_drowned_god_religion_trigger = yes
							any_demesne_title = {
								OR = {
									kingdom = {
										title = event_target:kingsmoot_title
									}
									empire = {
										title = event_target:kingsmoot_title
									}
								}
							}
						}
						OR = {
							is_female = no
							is_ruler = yes
							prestige = 500
							dynasty = PREV
							event_target:kingsmoot_title = {
								OR = {
									has_law = true_cognatic_succession
									has_law = enatic_cognatic_succession
									has_law = enatic_succession
								}
							}
							has_character_flag = kingsmoot_imposter #event character
						}

						NOT = { has_character_modifier = election_support_called }
						NOT = { has_character_flag = favor_vote_called }
					}
					ROOT = { save_event_target_as = favor_vote_caller }
					set_character_flag = favor_vote_called
					character_event = { id = ZE.8165 days = 1 random = 1 }
				}
			}
		}
	}

	option = { #Nothing
		name = "EVTOPTDkingsmoot.9"
		ai_chance = { factor = 1 }
		#set_character_flag = no_gift
	}

	option = { #Large Gifts
		name = "EVTOPTAkingsmoot.9"
		trigger = {
			wealth = 100
			NOT = { check_variable = { which = kingsmoot_gifts value = 401 } } #Max 500
		}
		ai_chance = {
			factor = 12
			modifier = {
				factor = 0.33
				trait = greedy
			}
			modifier = {
				factor = 0
				OR = {
					war = yes
					prisoner = yes
					wealth < 200
				}
			}
		}
		wealth = -100
		custom_tooltip = {
			text = TOOLTIPkingsmoot.9large_gift
			change_variable = { which = kingsmoot_gifts value = 100 }
		}
	}

	option = { #Regular Gifts
		name = "EVTOPTBkingsmoot.9"
		trigger = {
			wealth = 50
			NOT = { check_variable = { which = kingsmoot_gifts value = 451 } }
		}
		ai_chance = {
			factor = 6
			modifier = {
				factor = 0.66
				trait = greedy
			}
			modifier = {
				factor = 0
				OR = {
					war = yes
					prisoner = yes
					AND = {
						is_ruler = yes
						wealth < 100
					}
				}
			}
		}
		wealth = -50
		custom_tooltip = {
			text = TOOLTIPkingsmoot.9regular_gift
			change_variable = { which = kingsmoot_gifts value = 50 }
		}
		#set_character_flag = regular_gift
	}

	option = { #Withdraw fund
		name = "EVTOPTCkingsmoot.9"
		trigger = {
			check_variable = { which = kingsmoot_gifts value = 1 }
		}
		ai_chance = {
			factor = 3
			modifier = {
				factor = 0
				NOR = {
					war = yes
					prisoner = yes
					wealth < 0
				}
			}
			modifier = {
				factor = 1.5
				trait = greedy
			}
		}
		wealth = kingsmoot_gifts
		custom_tooltip = {
			text = TOOLTIPkingsmoot.9small_gift
			set_variable = { which = kingsmoot_gifts value = 0 }
		}
	}

	option = { #Sunset Sea
		name = "EVTOPTEkingsmoot.9"
		trigger = {
			OR = {
				has_landed_title = c_lonely_light
				dynasty = 114
				trait = traveller
			}
			OR = {
				trait = lunatic
				trait = mystic
				trait = ambitious
			}
			NOT = { has_character_flag = sunset_sea }
			NOT = { has_character_flag = threats }
			NOT = { has_character_flag = peace }
		}
		ai_chance = { factor = 3 }
		custom_tooltip = {
			text = TOOLTIPkingsmoot.9sunset_sea
		}
		set_character_flag = sunset_sea
	}

	option = { #Threaten
		name = "EVTOPTFkingsmoot.9"
		trigger = {
			OR = {
				higher_tier_than = BARON
				any_liege = {
					is_close_relative = ROOT
					primary_title = { has_law = succ_kingsmoot_elective }
				}
			}
			OR = {
				trait = lunatic
				trait = wroth
			}
			NOT = { trait = patient }
			NOT = { trait = kind }
			NOT = { trait = honorable }
			NOT = { trait = just }
			NOT = { trait = gregarious }
			NOT = { has_character_flag = sunset_sea }
			NOT = { has_character_flag = threats }
			NOT = { has_character_flag = peace }
		}
		ai_chance = { factor = 3 }
		custom_tooltip = {
			text = TOOLTIPkingsmoot.9threats
		}
		set_character_flag = threats
	}

	option = { #Peace
		name = "EVTOPTGkingsmoot.9"
		trigger = {
			diplomacy = 10
			OR = {
				NOT = { is_drowned_god_religion_trigger = yes }
				has_religion_feature = religion_peaceful
				has_religion_feature = religion_feature_nath_pagan
				trait = cynical
				pacifist = yes
			}
			NOT = { trait = ambitious }
			NOT = { has_character_flag = sunset_sea }
			NOT = { has_character_flag = threats }
			NOT = { has_character_flag = peace }
		}
		ai_chance = { factor = 3 }
		tooltip_info = diplomacy
		custom_tooltip = {
			text = TOOLTIPkingsmoot.9peace
		}
		set_character_flag = peace
	}
	after = {
		clr_character_flag = kingsmoot_tribute
	}
}
