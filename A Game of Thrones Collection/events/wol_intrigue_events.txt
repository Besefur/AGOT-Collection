# Schemer level up chance
character_event = {
	id = WoL.2000

	hide_window = yes

	is_triggered_only = yes
	has_dlc = "Way of Life"

	trigger = {
	}

	immediate = {
		# Chance of trait adjustments
		random_list = {
			50 = {
			}
			10 = {
				modifier = {
					factor = 0
					OR = {
						trait = honest
						trait = deceitful
						trait = honorable
						personality_traits = 6
					}
				}
				add_trait = deceitful
				character_event = { id = 38255 } # Deceitful notification
			}
			20 = {
				modifier = {
					factor = 0
					NOT = { trait = honest }
				}
				remove_trait = honest
				character_event = { id = 38311 } # Honest loss notification
			}
			20 = {
				modifier = {
					factor = 0
					NOT = { trait = honorable }
				}
				remove_trait = honorable
				character_event = { id = trait_notification.17 } # Honest loss notification
			}
		}

		# Chance of various Intrigue boosters
		character_event = { id = WoL.2001 days = 7 }

		# Chance of levelling up Schemer skills
		if = {
			limit = {
				has_character_modifier = wol_schemer_2
			}
			random = {
				chance = 10
				character_event = { id = WoL.2003 days = 4 } # Gain the Schemer or Impaler trait
			}
			break = yes
		}
		if = {
			limit = {
				has_character_modifier = wol_schemer_1
			}
			random = {
				chance = 20
				remove_character_modifier = wol_schemer_1
				add_character_modifier = {
					modifier = wol_schemer_2
					duration = -1
				}
			}
			break = yes
		}

		if = {
			limit = {
				NOT = {
					OR = {
						trait = schemer
						trait = impaler
					}
				}
			}
			random = {
				chance = 50
				add_character_modifier = {
					modifier = wol_schemer_1
					duration = -1
				}
			}
		}
	}
}
