character_event = {
	id = religious.1
	desc = "EVTDESCreligious.1"

	only_rulers = yes

	trigger = {
		demesne_size = 1
		OR = {
			is_theocracy = yes
			government = theocratic_feudal_government
		}
		OR = {
			AND = {
				is_drowned_god_religion_trigger = yes
				is_female = no
			}
			religion = the_seven
			religion = rhllor
			religion = rahlooism
			religion = harpy
			religion = qarth_warlocks
			religion = shadowbinders
			religion = bearded_priests
			religion = summer_rel
			AND = {
				OR = {
					religion = boash
					religion = boash_reformed
				}
				NOT = { trait = blinded }
				NOT = { trait = eunuch }
			}
			is_reformed_religion = yes
		}
		NOT = {
			has_priest_trait_trigger = yes
		}
	}

	mean_time_to_happen  = {
		days = 1
	}

	option = {
		name = "EVTOPTAreligious.1"
		if = {
			limit = {
				OR = {
					religion = boash
					religion = boash_reformed
				}
			}
			add_trait = blinded
			add_trait = eunuch
			break = yes
		}
		add_priest_trait_effect = yes
		recalc_succession = yes
	}
}
