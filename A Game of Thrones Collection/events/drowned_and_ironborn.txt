	id = ironborn_events.10
	title = EVTTITLEironborn_events.10
	desc = "EVTDESCironborn_events.10" #You have drowned!
	picture = "GFX_drowned"

	min_age = 16
	max_age = 45
	only_men = yes
	only_capable = yes
	prisoner = no

		is_drowned_god_religion_trigger = yes

			factor = 0.4

		}

			limit =  { trait = hedonist }
			remove_trait = hedonist
		}
		if = {
			limit =  { trait = lustful }
			remove_trait = lustful
		}
		if = {
			limit =  { trait = gluttonous }
			remove_trait = gluttonous
			lose_weight_medium_effect = yes
			weight_trait_check_effect = yes
		}
		if = {
			limit =  { trait = greedy }
			remove_trait = greedy
		}
		if = {
			limit =  { trait = cynical }
			remove_trait = cynical
		}
		if = {
			limit =  { trait = drunkard }
			remove_trait = drunkard
		}
		if = {
			limit =  { trait = slothful }
			remove_trait = slothful
		}
		if = {
			limit =  { trait = maester }

	trigger = {
		num_of_artifacts = 1
		any_artifact = {
			had_artifact_flag = { flag = flagship days = 14600 }
		}
	}

	immediate = {
		random_artifact = {
			limit = { had_artifact_flag = { flag = flagship days = 14600 } }
			save_event_target_as = flagship_to_retire
		}
	}

	option = {
		name = "EVTOPTAironborn_events.91"
		event_target:flagship_to_retire = {
			destroy_artifact = yes
		}
	}
	option = {
		name = "EVTOPTBironborn_events.91" #Renovate
		trigger = {
			OR = {
				ROOT = { ai = no }
				event_target:flagship_to_retire = {
					is_artifact_equipped = yes
					OR = {
						AND = {
							artifact_type = small_flagship
							ROOT = { wealth = 30 }
						}
						AND = {
							artifact_type = regular_flagship
							ROOT = { wealth = 60 }
						}
						AND = {
							artifact_type = enormous_flagship
							ROOT = { wealth = 115 }
						}
					}
				}
			}
		}
		event_target:flagship_to_retire = {
			clr_artifact_flag = flagship
			set_artifact_flag = flagship
			if = {
				limit = { artifact_type = enormous_flagship }
				ROOT = { wealth = -115 }
			}
			if = {
				limit = { artifact_type = regular_flagship }
				ROOT = { wealth = -60 }
			}
			if = {
				limit = { artifact_type = small_flagship }
				ROOT = { wealth = -30 }
			}
		}
	}
