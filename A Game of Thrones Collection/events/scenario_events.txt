			reverse_war = {
				target = ROOT

		}

			limit = { ai = no }
			chronicle = {
				entry = CHRONICLE_BLEEDING_YEARS
				picture = GFX_evt_volantis_2
			}
		}
		e_rebels = {
			persistent_event_target:chronicle_player = {
				if = {
					limit = {
						NOT = { character = ROOT }
						OR = {
							dynasty = ROOT
							is_liege_or_above = ROOT
							capital_scope = { region = world_free_cities }
						}
					}
					chronicle = {
						entry = CHRONICLE_BLEEDING_YEARS
						portrait = [Root.GetID]
					}
				}
			}
		}
	}

}
