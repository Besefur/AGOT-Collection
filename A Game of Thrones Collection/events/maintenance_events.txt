#Five yearly tyranny reduction
character_event = {
	id = maintenance.2

	is_triggered_only = yes
	hide_window = yes

	desc = EVTDESCmaintenance.2

	# trigger = {
		# check_variable = { which = "tyrant" value = 0.9 }
	# }

	immediate = {
		clr_character_flag = tyranny_decay_check
		if = {
			limit = {
				NOT = { check_variable = { which = "tyrant" value = 9.9 } }
			}
			set_variable = { which = "tyrant" value = 0 }
		}
		if = {
			limit = {
				check_variable = { which = "tyrant" value = 9.9 }
			}
			change_variable = { which = "tyrant" value = -5 }
			set_character_flag = tyranny_decay_check
			character_event = { id = maintenance.2 days = 1825 } #Check again in five years
		}
		if = { #recalculate tyranny trait
			limit = { NOT = { trait = tyrannical_stain } }
			character_event = { id = maintenance.1 }
		}
		if = {
			limit = {
				trait = tyrannical_stain
				NOT = { check_variable = { which = "tyrant" value = 0.9 } }
			}
			remove_trait = tyrannical_stain
		}
	}

	option = {
		name = "OK"
	}
}

	id = maintenance.69
	desc = "EVTDESCmaintenance.69"

	is_triggered_only = yes
	only_rulers = yes

				has_title_flag = ze_dyn_merc
			}
		}

	}

	immediate = {
		export_to_variable = { which = current_wealth value = wealth }

		#Calculate wealth to subtract (expenses, equipment etc)
		set_variable = { which = expenses which = current_wealth }
		multiply_variable = { which = expenses value = 0.33 } #33%

		#subtract the expenses
		subtract_variable = { which = current_wealth which = expenses }

		#Count number of courtiers eligible for a share
		set_variable = { which = merc_courtier value = 8 }	#Leader gets 8 shares
		any_courtier = {
			limit = {
				liege = { character = ROOT }
				is_ruler = no
				age = 16
				prisoner = no
				is_incapable = no
				NOT = { trait = slave }
				NOT_mythical_creature_trigger = yes
			}
			set_character_flag = eligible_for_share
			ROOT = { change_variable = { which = merc_courtier value = 1 } }
			#Councillors/commanders get extra shares
			if = {
				limit = {
					OR = {
						is_councillor = yes
						is_voter = yes
					}
				}
				ROOT = { change_variable = { which = merc_courtier value = 2 } }
			}
			if = {
				limit = { has_job_title = job_marshal }
				ROOT = { change_variable = { which = merc_courtier value = 1 } }
			}
			if = {
				limit = { has_job_title = job_castellan }
				ROOT = { change_variable = { which = merc_courtier value = 1 } }

				limit = {
					has_minor_title = title_commander
					NOR = {
						is_councillor = yes
						is_voter = yes
					}
				}
				ROOT = { change_variable = { which = merc_courtier value = 1 } }
			}
		}

		#Calculate value of a share
		divide_variable = { which = current_wealth which = merc_courtier }
		any_courtier = {
			limit = { has_character_flag = eligible_for_share }
			set_variable = { which = current_wealth which = ROOT }
		}

	}

		name = OK
		custom_tooltip = {
			text = TOOLTIPmaintenance.69EXPENSES
			hidden_tooltip = {
				multiply_variable = { which = expenses value = -1 }
				wealth = expenses
				set_variable = { which = expenses value = 0 }
			}
		}
		custom_tooltip = {
			text = TOOLTIPmaintenance.69SHARES
			hidden_tooltip = {
				multiply_variable = { which = current_wealth value = -1 }
				any_courtier = {
					limit = { has_character_flag = eligible_for_share }
					wealth = current_wealth
					ROOT = { wealth = current_wealth }
					#Councillors/commanders get extra shares
					if = {
						limit = {
							OR = {
								is_councillor = yes
								is_voter = yes
							}
						}
						wealth = current_wealth
						wealth = current_wealth
						ROOT = {
							wealth = current_wealth
							wealth = current_wealth
						}
					}

						limit = { has_job_title = job_castellan }
