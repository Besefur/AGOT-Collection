namespace = congenitalcrown
character_event = {
	id = congenitalcrown.0 #Durrandon/Baratheon

	is_triggered_only = yes
	hide_window = yes

	trigger = {
		any_owned_bloodline = {
			has_bloodline_flag = bloodline_baratheon_armour
		}
		can_wear_a_crown_trigger = yes
		NOR = {
			has_global_flag = congenital_crown_1
			has_character_flag = congenital_crown
		}
	}
	immediate = {
		add_artifact = congenital_crown_1
		new_artifact = {
			set_artifact_flag = heirloom_@PREV
			set_artifact_flag = title_heirloom_@PREV
			set_artifact_flag = heirloom_@event_target:ROOT_primary_title
		}
		set_global_flag = congenital_crown_1
		set_character_flag = congenital_crown
	}
}
