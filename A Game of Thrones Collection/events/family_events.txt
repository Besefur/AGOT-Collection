									has_law = enatic_cognatic_succession
									has_law = enatic_succession
								}
							}
						}
					}
				}
			}
			set_character_flag = valid_heir
			set_character_flag = received_inheritance
			ROOT = {
				change_variable = { which = "num_of_valid_heirs_child" value = 1 }
			}
			event_target:current_heir_wealth_split  = {
				change_variable = { which = "num_of_valid_heirs_total" value = 1 }
				PREV = { save_event_target_as = wealth_split_example }
			}
		}
		any_child_even_if_dead = { #Grandchildren (only if parent is dead)
			limit = {
				is_alive = no
				NOT = { ROOT = { current_heir = { character = PREVPREV } } }
				can_inherit_titles_trigger = yes
				OR = { #is unlanded, or child is the heir
					lower_tier_than = COUNT
					ROOT = { current_heir = { is_child_of = PREVPREV } }
				}
				#Heirs of wrong gender must be unmarried
				OR = {
					NOT = {
						any_spouse_even_if_dead = {
							OR = {
								is_alive = no
								is_alive = yes
							}
						}
					}
					ROOT = { primary_title = { has_law = true_cognatic_succession } }
					AND = {
						is_female = no
						ROOT = {
							primary_title = {
								OR = {
									has_law = cognatic_succession
									has_law = agnatic_succession
								}
							}
						}
					}
					AND = {
						is_female = yes
						ROOT = {
							primary_title = {
								OR = {
									has_law = enatic_cognatic_succession
									has_law = enatic_succession
								}
							}
						}
					}
				}
			}
			any_child = {
				limit = {
					is_alive = yes
					NOT = { is_child_of = ROOT }
					NOT = { has_character_flag = valid_heir }
					NOT = { ROOT = { current_heir = { character = PREVPREV } } }
					lower_tier_than = COUNT
					can_inherit_titles_trigger = yes
					OR = {
						prisoner = no
						NOT = { host = { character = ROOT } }
						NOT = { has_character_flag = lost_trial_by_combat }
					}
					OR = {
						same_realm = ROOT
						NOT = { has_character_flag = exiled }
					}
					#Heirs of wrong gender must be unmarried
					OR = {
						NOT = {
							any_spouse_even_if_dead = {
								OR = {
									is_alive = no
									is_alive = yes
								}
							}
						}
						ROOT = { primary_title = { has_law = true_cognatic_succession } }
						AND = {
							is_female = no
							ROOT = {
								primary_title = {
									OR = {
										has_law = cognatic_succession
										has_law = agnatic_succession
									}
								}
							}
						}
						AND = {
							is_female = yes
							ROOT = {
								primary_title = {
									OR = {
										has_law = enatic_cognatic_succession
										has_law = enatic_succession
									}
								}
							}
						}
					}
				}
				set_character_flag = valid_heir
				set_character_flag = received_inheritance
				event_target:current_heir_wealth_split  = {
					change_variable = { which = "num_of_valid_heirs_total" value = 1 }
					PREV = { save_event_target_as = wealth_split_example }
				}
			}
		}
		any_child_even_if_dead = { #Add number of dead child lines to child count
			limit = {
				is_alive = no
				any_child = { has_character_flag = valid_heir }
			}
			ROOT = {
				change_variable = { which = "num_of_valid_heirs_child" value = 1 }
			}
		}


		if = {
			limit = { check_variable = { which = "num_of_valid_heirs_child" value = 1 } }
			#share depends on number of children, divide 3 years worth of income
			#divide equallt between each child's family, even if dead
			#Grandchildren then split their parent's share
			set_variable = { which = heir_wealth_share value = 3 }
			divide_variable = { which = heir_wealth_share which = num_of_valid_heirs_child }
			any_child = { #Living children
				limit = { has_character_flag = valid_heir }
				ROOT = {
					transfer_scaled_wealth = {
						to = PREV
						value = heir_wealth_share
					}
				}
				clr_character_flag = valid_heir
			}
			any_child_even_if_dead = { #Grandchildren (Parent is dead)
				limit = {
					is_alive = no
					any_child = { has_character_flag = valid_heir }
				}
				ROOT = {
					set_variable = { which = "num_of_valid_heirs_grandchild" value = 0 }
					set_variable = { which = heir_wealth_share_grandchild which = heir_wealth_share }
				}
				any_child = {
					limit = { has_character_flag = valid_heir }
					ROOT = {
						change_variable = { which = "num_of_valid_heirs_grandchild" value = 1 }
					}
				}
				ROOT = {
					divide_variable = { which = heir_wealth_share_grandchild which = num_of_valid_heirs_grandchild }
				}
				any_child = {
					limit = { has_character_flag = valid_heir }
					ROOT = {
						transfer_scaled_wealth = {
							to = PREV
							value = heir_wealth_share_grandchild
						}
					}
					clr_character_flag = valid_heir
				}
			}
			event_target:current_heir_wealth_split = {  #Inform player heir
				ROOT = {
					export_to_variable = {
					   which = wealth_split
					   value = yearly_income
					   who = THIS
					}
				}
				set_variable = { which = wealth_split which = ROOT }
				multiply_variable = { which = wealth_split value = 3 }
				character_event = { id = 37354 days = 1 }
			}
		}
		else = {
			event_target:current_heir_wealth_split = {
				set_variable = { which = num_of_valid_heirs_total value = 0 }
			}
		}
	}

	option = {
		name = OK
	}
}
