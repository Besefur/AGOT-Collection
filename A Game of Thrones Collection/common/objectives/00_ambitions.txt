obj_befriend_liege = {
	type = character

	potential = {
		ai = no
		NOT_mythical_creature_trigger = yes
		liege = { NOT = { character = ROOT } }
	}
	allow = {
		is_adult = yes
		is_incapable = no
		liege = {
			is_adult = yes
			is_incapable = no
			opinion = { who = ROOT value = 10 }
			reverse_opinion = { who = ROOT value = 10 }
			NOT = { is_friend = ROOT }
			NOT = { is_rival = ROOT }
			NOT = { is_child_of = ROOT }
			NOT = { is_parent_of = ROOT }
			NOT = { is_married = ROOT }
			NOT = { opinion = { who = ROOT value = 60 } }
		}
	}
	chance = {
		factor = 0
	}
	success = {
		liege = {
			NOT = { character = ROOT }
			OR = {
				opinion = { who = ROOT value = 60 }
				is_friend = ROOT
			}
		}
	}
	abort = {
		OR = {
			has_character_flag = abort_obj
			trait = incapable
			liege = {
				OR = {
					character = ROOT
					is_rival = ROOT
					trait = incapable
					is_married = ROOT
					is_adult = no
					is_parent_of = ROOT
					NOT = { opinion = { who = ROOT value = -50 } }
					NOT = { reverse_opinion = { who = ROOT value = -50 } }
				}
			}
		}
	}
	abort_effect = {
		clr_character_flag = abort_obj
		clr_character_flag = obj_check
		clr_character_flag = obj_befriend_liege_event
		clr_character_flag = obj_befriend_liege_wine_event
		clr_character_flag = obj_befriend_liege_wenching_event
	}
	effect = {
		set_character_flag = obj_befriend_liege
		clr_character_flag = obj_befriend_liege_event
		clr_character_flag = obj_befriend_liege_wine_event
		clr_character_flag = obj_befriend_liege_wenching_event
		liege = {
			hidden_tooltip = {
				if = {
					limit = { NOT = { is_friend = ROOT } }
					character_event = { id = ambition.700 } #inform liege
				}
			}
			tooltip = { add_friend = ROOT	}
		}
	}
}
