				text = TOOLTIPJOGOSWARPROHIBITION

		}

			FROM = { culture = ROOT } #target is same culture
			ROOT = { #has unified the nomads

					has_character_modifier = greatest_of_khans

	can_use_title = {
		universal_cb_restrictions_trigger = yes
		# Dothraki/Jogos Nhai Only
		ROOT = {
			is_nomadic = yes
			NAND = {
				has_landed_title = c_dosh_khaleen
				trait = dosh_khaleen
			}
			top_liege = {
				is_nomadic = yes
			}
		}
		FROM = {
			has_landed_title = PREV
			NAND = {
				has_landed_title = c_dosh_khaleen
				trait = dosh_khaleen
			}
			NOT = { same_realm = ROOT }
		}
		is_primary_holder_title = yes
		controls_religion = no
		NOT = { title = d_nightswatch }
		NOT = { title = d_kingsguard }
		# Can only invade neighbors
		ROOT = {
			any_realm_province = {
				has_owner = yes
				owner = {
					OR = {
						character = ROOT
						is_liege_or_above = ROOT
					}
				}
				any_neighbor_province = {
					has_owner = yes
					owner = {
						OR = {
							character = FROM
							is_liege_or_above = FROM
						}
					}
				}
			}
		}
		ROOT = { NOT = { primary_title = { title = d_kingsguard } } }
		FROM = { NOT = { primary_title = { title = d_kingsguard } } }
		FROM = { NOT = { primary_title = { title = d_nightswatch } } }
	}

	is_valid_title = {
		universal_cb_valid_restrictions_trigger = yes
		# Dothraki Only
		ROOT = {
			is_nomadic = yes
			top_liege = {
				is_nomadic = yes
			}
		}
		FROM = { has_landed_title = PREV }
		OR = {
			FROM = {
				OR = {
					culture = ROOT
					is_nomadic = yes
				}
			}
			ROOT = {
				independent = yes
				OR = {
					has_character_modifier = greatest_of_khans
					has_character_modifier = jhattar_of_jogos_nhai
				}
			}
		}
	}

	on_add_title = {
		hidden_tooltip = { FROM = { character_event = { id = diplomatic.3099 } } } #call for aid?
		e_rebels = {
			persistent_event_target:chronicle_player = {
				if = {
					limit = { character = ROOT }
					chronicle = {
						entry = CHRONICLE_WAR_INVASION_ATTACKER
						portrait = [From.GetID]
					}
					break = yes
				}
				if = {
					limit = { character = FROM }
					chronicle = {
						entry = CHRONICLE_WAR_INVASION_DEFENDER
						portrait = [Root.GetID]
					}
					break = yes
				}
				if = {
					limit = { is_liege_or_above = ROOT }
					chronicle = {
						entry = CHRONICLE_WAR_INVASION_ATTACKER_LIEGE
						portrait = [Root.GetID]
					}
					break = yes
				}
				if = {
					limit = { is_liege_or_above = FROM }
					chronicle = {
						entry = CHRONICLE_WAR_INVASION_DEFENDER_LIEGE
						portrait = [From.GetID]
					}
				}
			}
		}
	}

	on_success_title = {
		hidden_tooltip = {
			holder_scope = { save_event_target_as = old_usurped_title_holder }
			ROOT = { add_character_modifier = { name = victory_timer duration = 3 } }
			ROOT = { #Improve Clan Opinion
				clan_title = {
					save_event_target_as = liege_clan
				}
				any_vassal = {
					limit = { is_nomadic = yes }
					clan_title = {
						clan_opinion = {
							who = event_target:liege_clan
							modifier = opinion_won_nomad_war
							years = 2
						}
					}
				}
			}
		}
		if = { #If enemy is not nomad, and has no dothraki sea territory, subjugate
			limit = {
				FROM = {
					NOT = { is_nomadic = yes }
					NOT = { any_realm_province = { traditional_nomad_territory_trigger = yes } }
				}
			}
			hidden_tooltip = {
				if = {
					limit = { ROOT = { tier = KING } }
					FROM = {
						any_vassal = {
							limit = { tier = DUKE }
							set_defacto_liege = ROOT
						}
						if = {
							limit = { government = alliance_government } #disband alliance
							set_character_flag = conquest
							character_event = { id = free_cities.15 tooltip = TOOLTIPfree_cities.15 }
						}
						any_demesne_title = {
							limit = {
								tier = KING
								controls_religion = no
							}
							usurp_title = { target = ROOT type = invasion }
						}
					}
				}
				if = {
					limit = { ROOT = { tier = EMPEROR } }
					FROM = {
						any_vassal = {
							limit = { tier = KING }
							set_defacto_liege = ROOT
						}
						if = {
							limit = { government = alliance_government } #disband alliance
							set_character_flag = conquest
							character_event = { id = free_cities.15 tooltip = TOOLTIPfree_cities.15 }
						}
						any_demesne_title = {
							limit = {
								tier = EMPEROR
								controls_religion = no
							}
							usurp_title = { target = ROOT type = invasion }
						}
					}
				}
			}
			FROM = {
				set_defacto_liege = ROOT
				hidden_tooltip = { character_event = { id = maintenance.25 } }#Subjugation flag maintenance
			}
		}
		if = { #If enemy is not nomad, and has dothraki sea territory, sack occupied land
			limit = {
				FROM = {
					NOT = { is_nomadic = yes }
					any_realm_province = { traditional_nomad_territory_trigger = yes }
				}
			}
			custom_tooltip = {
				text = dothraki_invasion_succ_tipB
				hidden_tooltip = {
					FROM = {
						any_realm_province = {
							limit = {
								any_province_holding = {
									controller = {
										OR = {
											character = ROOT
											is_liege_or_above = ROOT
										}
									}
									NOT = { holding_type = trade_post }
								}
							}
							random_province_holding = { #spoils usually goes to nomad controller
								limit = {
									controller = {
										OR = {
											character = ROOT
											is_liege_or_above = ROOT
										}
									}
									NOT = { holding_type = trade_post }
								}
								controller = {
									if = {
										limit = { is_nomadic = yes }
										save_event_target_as = nomad_pillager
									}
									if = {
										limit = { is_nomadic = no }
										ROOT = { save_event_target_as = nomad_pillager }
									}
								}
							}
							if = { #gain slaves and gold, liege gets 50%
								limit = {
									NOT = { has_province_modifier = rampaging_nomads }
								}
								if = {
									limit = { any_province_holding = { base_value_6_trigger = yes } }
									event_target:nomad_pillager = {
										wealth = 125
										character_event = { id = slavery.1000 tooltip = "EVTTOOLTIPSLAVERY1000" }
									}
									ROOT = { wealth = 125 }
								}
								if = {
									limit = {
										any_province_holding = { base_value_5_trigger = yes }
										NOT = { any_province_holding = { base_value_6_trigger = yes } }
									}
									event_target:nomad_pillager = {
										wealth = 100
										character_event = { id = slavery.1000 tooltip = "EVTTOOLTIPSLAVERY1000" }
									}
									ROOT = { wealth = 100 }
								}
								if = {
									limit = {
										any_province_holding = { base_value_4_trigger = yes }
										NOT = { any_province_holding = { base_value_5_trigger = yes } }
									}
									event_target:nomad_pillager = { wealth = 75 }
									ROOT = { wealth = 75 }
								}
								if = {
									limit = {
										any_province_holding = { base_value_3_trigger = yes }
										NOT = { any_province_holding = { base_value_4_trigger = yes } }
									}
									event_target:nomad_pillager = { wealth = 50 }
									ROOT = { wealth = 50 }
								}
								if = {
									limit = {
										any_province_holding = { base_value_2_trigger = yes }
										NOT = { any_province_holding = { base_value_3_trigger = yes } }
									}
									event_target:nomad_pillager = { wealth = 25 }
									ROOT = { wealth = 25 }
								}
								if = {
									limit = {
										any_province_holding = { base_value_1_trigger = yes }
										NOT = { any_province_holding = { base_value_2_trigger = yes } }
									}
									event_target:nomad_pillager = { wealth = 13 }
									ROOT = { wealth = 12 }
								}
								if = {
									limit = {
										NOT = { any_province_holding = { base_value_1_trigger = yes } }
									}
									event_target:nomad_pillager = { wealth = 5 }
									ROOT = { wealth = 5 }
								}
							}
							if = { #dothraki sea territory is taken
								limit = { traditional_nomad_territory_trigger = yes }
								owner = {
									any_courtier = {
										limit = {
											location = { province_id = PREVPREVPREV }
											NOR = {
												trait = in_hiding
												trait = wikid
												trait = on_reaving
											}
											prisoner = no
											NOT_mythical_creature_trigger = yes
											NOT = { character = ROOT }
											NOT = { is_child_of = ROOT }
											NOT = { father = { is_child_of = ROOT } }
											NOT = { mother = { is_child_of = ROOT } }
											NOT = { character = event_target:nomad_pillager }
											NOT = { is_child_of = event_target:nomad_pillager }
											NOT = { father = { is_child_of = event_target:nomad_pillager } }
											NOT = { mother = { is_child_of = event_target:nomad_pillager } }
										}
										imprison = event_target:nomad_pillager
									}
								}
								county = { gain_title = event_target:nomad_pillager }
								province_event = { id = dothraki.610 } #province maintenance
							}
							if = { #non-dothraki sea territory is just sacked
								limit = { NOT = { traditional_nomad_territory_trigger = yes } }
								county = {
									any_de_jure_vassal_title = {
										limit = {
											controller = {
												OR = {
													character = ROOT
													is_liege_or_above = ROOT
												}
											}
										}
										add_holding_modifier = {
											modifier = recently_burnt_the_land
											months = 7
										}
									}
								}
								add_province_modifier = {
									modifier = rampaging_nomads
									years = 10
								}
								add_province_modifier = {
									modifier = burnt_the_land
									years = 10
								}
								if = {
									limit = { has_dlc = "Reapers" }
									if = {
										limit = { check_variable = { which = prosperity_value value = 50 } }
										change_variable = { which = prosperity_value value = -50 }
									}
									if = {
										limit = { NOT = { check_variable = { which = prosperity_value value = 50 } } }
										set_variable = { which = prosperity_value value = 0 }
									}
									if = {
										limit = {
											has_province_modifier = prosperity_modifier_3
											NOT = { check_variable = { which = prosperity_value value = 500 } }
										}
										remove_province_modifier = prosperity_modifier_3
										add_province_modifier = { name = prosperity_modifier_2 duration = -1 }
									}
									if = {
										limit = {
											has_province_modifier = prosperity_modifier_2
											NOT = { check_variable = { which = prosperity_value value = 250 } }
										}
										remove_province_modifier = prosperity_modifier_2
										add_province_modifier = { name = prosperity_modifier_1 duration = -1 }
									}
									if = {
										limit = {
											has_province_modifier = prosperity_modifier_1
											NOT = { check_variable = { which = prosperity_value value = 100 } }
										}
										remove_province_modifier = prosperity_modifier_1
									}
								}
							}
						}
					}
				}
			}
		}
		if = { #If enemy is fellow nomad of same culture, absorb
			limit = {
				FROM = {
					is_nomadic = yes
					culture = ROOT
				}
			}
			custom_tooltip = {
				text = dothraki_invasion_succ_tip
				hidden_tooltip = {
					FROM = { #Subjugate any vassals
						any_vassal = {
							limit = {
								lower_tier_than = ROOT
								NOT = { tier = BARON }
							}
							any_demesne_title = { #take any occupied titles
								limit = {
									tier = BARON
									controller = {
										OR = {
											character = ROOT
											is_liege_or_above = ROOT
										}
									}
								}
								county = { usurp_title = { target = ROOT type = invasion } }
							}
							set_defacto_liege = ROOT
						}
					}
					ROOT = { #take rest of occupied titles
						occupy_minors_of_occupied_settlements = FROM
						gain_all_occupied_titles = { who = FROM type = invasion }
					}
					FROM = { #Take high tier titles
						if = {
							limit = { ROOT = { tier = KING } }
							any_demesne_title = {
								limit = {
									tier = KING
									controls_religion = no
									NOT = { holder_scope = { clan_title = { title = PREVPREV } } }
								}
								usurp_title = { target = ROOT type = invasion }
							}
						}
						if = {
							limit = { ROOT = { tier = EMPEROR } }
							any_demesne_title = {
								limit = {
									tier = EMPEROR
									controls_religion = no
								}
								destroy_landed_title = THIS
							}
							any_demesne_title = {
								limit = { tier = EMPEROR }
								usurp_title = { target = ROOT type = invasion }
							}
						}
						set_defacto_liege = ROOT
						hidden_tooltip = { reverse_opinion = { who = ROOT modifier = opinion_defeated_khal } } #can freely executed defeated khals
					}
				}
			}
			FROM = { imprison = ROOT clr_character_flag = captured_in_battle }
		}
		if = { #If enemy is nomad of different culture, make tributary
			limit = {
				FROM = {
					is_nomadic = yes
					NOT = { culture = ROOT }
				}
			}
			ROOT = { make_tributary = { who = FROM tributary_type = default } }
		}
		ROOT = {
			hidden_tooltip = {
				character_event = { id = maintenance.21 } #Take dejure vassals (AGOT)
			}
		}
		any_attacker = {
			participation_scaled_prestige = 200
		}
		FROM = {
			prestige = -200
		}
		clear_event_target = old_usurped_title_holder


	}

	on_success = {
		#To handle Warrior Lodge mission "Wage War" completion...
		attacker = { save_event_target_as = attacker_target }
		defender = { save_event_target_as = defender_target }

		if = {
			limit = {
				event_target:attacker_target = { has_quest = quest_warrior_lodge_war }
				is_quest_target_of = event_target:attacker_target
			}
			event_target:attacker_target = {
				show_scope_change = no
				tooltip = { clr_quest = quest_warrior_lodge_war }
				set_character_flag = won_war@event_target:defender_target
				#character_event = { id = HF.10514 days = 3 } #Sent from on_action instead
			}
		}
	}

	on_fail = {
		ROOT = {
			prestige = -100
		}
		any_defender = {
			participation_scaled_prestige = 50
		}
	}

	on_reverse_demand = {
		hidden_tooltip = { FROM = { add_character_modifier = { name = victory_timer duration = 3 } } }
		ROOT = {
			prestige = -200
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
		}
		any_defender = {
			participation_scaled_prestige = 100
		}
	}

	attacker_ai_victory_worth = {
		factor = 100
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}

	defender_ai_defeat_worth = {
		factor = 100
	}

	# ai importance placed on this CB: scope is the targeted title, ROOT is the attacking character, FROM is the defending character
	ai_will_do = {
		factor = 2
		#Dragon intimidation
		modifier = {
			factor = 0
			ROOT = {
				NOT = { can_use_dragon_abilities_trigger = yes }
				NOT = { relative_power = { who = FROM power = 6 } }
			}
			FROM = {
				can_use_dragon_abilities_trigger = yes
				OR = {
					any_friend = { trait = dragon martial = 60 }
					AND = {  graphical_culture = dragongfx martial = 60 }
				}
				is_incapable = no
			}
		}
	}
